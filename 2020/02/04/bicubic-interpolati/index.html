源于科学计算课程作业，现将代码整理如下：

/*******2019.03.31**********
双三次多项式插值 Bi-cubic interpolation
插值数据大小384*122，网格间距24m*24m

122行，384列。。。。 
将数据模型插值为网格间距为dx=4m，dz=6m的数据模型。
数据文件名为md.txt
***************************/
#include<cstdio>
#include<canshu.h>
 //参数及常量定义 
 const int M=122,N=384;                       //原始数据大小
 const int dX=24,dZ=24,dz=6,dx=4;             //原始和插入后的网格间距 
 const int m=dZ/dz*(M-1)+1,n=dX/dx*(N-1)+1;  //插值后网格大小 
 double indata[M][N]={0.0};           //原始数据
 double outdata[m][n]={0.0}; 
 FILE * fpin,*fpout;         //两个文件指针，表示输入输出 

 struct matrix {  //定义矩阵结构体 
     int dx,dy;
     double **A;
     matrix()  {}
     matrix (int m ,int n) {  //动态申请空间 
         A=new double *[m];
         for(int i=0;i<m;i++) {
             A[i]=new double [n];
         }
         dx=n;
         dy=m;
     }
 }; 
 matrix operator * (const matrix c1,const matrix c2) { //定义矩阵乘法 
     matrix temp(c1.dy,c2.dx);
     double sum=0.0;
     if(c1.dx != c2.dy ) {  // 矩阵不合法 ，退出 
     printf("error!\n矩阵维度不匹配\n");
     return temp;
     }      
     for(int i=0;i<c1.dy;i++) {
         for(int u=0;u<c2.dx;u++) {        //c2的列 
             for(int j=0;j<c1.dx;j++) {
               sum+=1.0*c1.A[i][j]*c2.A[j][u] ;    
             }
             temp.A[i][u]=sum;
             sum=0.0;
         }
     }
     return temp;
 }
 matrix trans( const matrix  c) {          //矩阵转置 
     matrix temp(c.dx,c.dy);
     for(int i=0;i<c.dy;i++){
         for(int j=0;j<c.dx;j++) {
             temp.A [j][i]=c.A [i][j];
         }
     }
     return temp;
 }

double Dd[4][4]={{-1,3,-3,1},{3,-6,3,0},{-3,0,3,0},{1,4,1,0}}; 
matrix D(4,4) ;

void calculate( ) {
    double u,w;                  //  定义Δλ 
    printf("\n正在计算\n"); 
    for(int j=dZ/dz;j<m-dZ/dz-1;j++) {  //行号 
              w=((j*dz)%dZ)*1.0/(dZ*1.0);              //设置矩阵Z 
            matrix Z(1,4);
            Z.A[0][3]=1.0;
            for(int k=2;k>=0;k--) {
                Z.A[0][k]=Z.A[0][k+1]*1.0*w;
            }
       for(int i=dX/dx;i<n-dX/dx-1;i++) {  //  列 号 

            u=((i*dx)%dX)*1.0/(dX*1.0);                  //设置矩阵X 
            matrix X(1,4);
            X.A[0][3]=1.0;
            for(int k=2;k>=0;k--) {
            X.A[0][k]=X.A[0][k+1]*1.0*u;
            }

        int xx,yy;                     //xx,yy为原始数据对应下标 
        xx=i*dx/dX-1;yy=j*dz/dZ-1;
        matrix C(4,4);                 //设置矩阵C 
            for(int ii=0;ii<4;ii++) {      //ii为行数 
                for(int jj=0;jj<4;jj++) {
                    C.A[ii][jj]=indata[yy][xx]; // yy为行号 
                    xx++;
                }
                yy++;
                xx=i*dx/dX-1;
            }
            outdata[j][i]=(X*D*C*trans(D)*trans(Z)).A[0][0]/36.0;
        }
    } 
    //写入文件 
    printf("\n运算完成，开始写入\n");

    for(int j=dZ/dz;j<m-dZ/dz-1;j++) {              //j代表行号 
        for(int i=dX/dx;i<n-dX/dx-2;i++) {        //i代表列号 
          fprintf(fpout,"%lf ",outdata[j][i]);
        }
        fprintf(fpout,"%lf\n",outdata[j][n-dX/dx-2]);
    }
}




 int main() {
     printf("**********开始执行*********\n");
     //给D矩阵赋值 
   for(int i=0;i<4;i++) {
     for (int j=0;j<4;j++) {
        D.A[i][j]=Dd[i][j];
     }
 }    

   char *filepath,*filepath1,*filepath2;             //打开文件 
  filepath="canshu.txt";
  fpin=fopen(filepath,"r");
  fscanf(fpin,"%s %s",filepath1,filepath2);    
  fscanf(fpin,"%d %d %d %d",&dX,&dZ,&dx,&dz);
  fclose(fpin);


  fpin=fopen(filepath1,"r");
  printf("\n开始导入数据\n");
  for(int i=0;i<M;i++) {      // 行 
      for(int j=0;j<N;j++ ) {   // 列 
          fscanf(fpin,"%lf",&indata[i][j]);   
      }
  }
  fclose(fpin);               //输入完成 
  printf("\n数据导入成功！\n");

  fpout=fopen("output.txt","w");
  calculate();         //计算插值 
  fclose(fpout);
  printf("\n*********程序运行结束**********");
     return 0;
 }
